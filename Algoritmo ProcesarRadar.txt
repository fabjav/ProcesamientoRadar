Algoritmo ProcesarRadar

// 1. Cargar templates
LEER matriz_MA_mendoza
LEER matriz_MA_sanra

// 2. Definir configuraciones
radar_mendoza:
	carpeta_entrada <- 'ruta/imagenes_mendoza'
	carpeta_salida <- 'ruta/salida/mendoza'
	matriz_MA <- matriz_MA_mendoza

radar_sanra:
	carpeta_entrada <- 'ruta/imagenes_sanra'
	carpeta_salida <- 'ruta/salida/sanra'
	matriz_MA <- matriz_MA_sanra

// 3. Procesar cada radar
PARA CADA radar EN [radar_mendoza, radar_sanra] HACER

	// Verificar carpeta de salida
	SI radar.carpeta_salida NO EXISTE ENTONCES
		CREAR radar.carpeta_salida
	FIN SI

	// Procesar imágenes
	PARA CADA archivo EN radar.carpeta_entrada HACER
		
		SI archivo es .gif ENTONCES
			
			nombre_salida <- generar nombre
			
			SI nombre_salida NO EXISTE EN radar.carpeta_salida ENTONCES
				
				imagen <- LEER archivo
				
				// --- PROCESAMIENTO ---
				
				// 1. Recorte
				Recortar bordes de la imagen
				
				// 2. Selección de tormenta
				Para cada pixel:
					SI color pertenece a rango DBZ ENTONCES
						conservar pixel
					SINO
						reemplazar por fondo
				
				// 3. Eliminación y reconstrucción de elementos finos (textos y líneas)
				Para cada pixel (i,j):

					SI pixel pertenece a estructura fina (grosor ~1px)
					Y color NO pertenece al rango DBZ ENTONCES
						
						marcar pixel como "a reconstruir"

				FIN PARA

				// Relleno de esos pixeles
				Para cada pixel (i,j) marcado HACER
					
					obtener vecinos (4 u 8 conectividad)
					seleccionar vecinos válidos (que sean tormenta)
					
					determinar color dominante entre vecinos
					
					reemplazar pixel (i,j) con color dominante

				FIN PARA
				// 4. Reconstrucción de la marca de agua (DACC)
				Para cada pixel (i,j) donde radar.matriz_MA(i,j) = 1 HACER
					
					SI pixel (i,j) pertenece a tormenta ENTONCES
						
						obtener vecinos (4 u 8 conectividad)
						seleccionar vecinos válidos (que sean tormenta)
						
						determinar color dominante
						
						reemplazar pixel (i,j) con color dominante
					
					FIN SI

				FIN PARA
				
				// --- FIN PROCESAMIENTO ---
				
				GUARDAR imagen en radar.carpeta_salida
				
			FIN SI
			
		FIN SI
		
	FIN PARA

FIN PARA

FIN Algoritmo
