Algoritmo ProcesarRadar

// 1. Cargar Template de marca de agua
LEER matriz_MA_dacc

// 2. Definir carpetas
carpeta_entrada <- 'ruta/imagenes'
carpeta_salida <- 'ruta/salida'

// 3. Verificar carpeta de salida
SI carpeta_salida NO EXISTE ENTONCES
	CREAR carpeta_salida
FIN SI

// 4. Procesar imágenes
PARA CADA archivo EN carpeta_entrada HACER

	SI archivo es .gif ENTONCES
		
		nombre_salida <- generar nombre archivo
		
		SI nombre_salida NO EXISTE EN carpeta_salida ENTONCES
			
			imagen <- LEER archivo
			
			// Procesamiento
			
			Recortar bordes
			
			Seleccionar tormenta según colores DBZ 
			y eliminar fondo (mantener tormenta + MA sobre tormenta)
			
			Eliminar textos:
				- grosor aproximado 1px
				- color en rango negro/gris
			
			Reconstruir tormenta en zona MA:
				- usar matriz_MA_dacc
				- identificar píxeles de MA
				- si están sobre tormenta:
					reemplazar por color dominante de vecinos
			
			GUARDAR imagen en carpeta_salida
			
		FIN SI
		
	FIN SI

FIN PARA

FIN Algoritmo
